3. 编写后序遍历二叉树的非递归算法

   ```c
   void PostOrder(BiTree T){
       InitStack(S);
       p=T;
       r=NULL;
       while(p || IsEmpty(S)){
           if(p){
               push(S,p);
               p = p->lchild;
           }else{
               GetTopp(S,p);
           }
           if(p->rchild && p->rchild!=r){
               p=p->rchild;
               push(S,p);
               p=p->lchild;
           }else{
               pop(S,p);
               visit(p->data);
               r=p;
               p=null;
           }
       }
   }
   ```
