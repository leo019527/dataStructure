6. 有一个带头结点的单链表L，设计一个算法使其元素递增有序。

   ```c
   void sortLinkedList(Node *l){
       //选择排序
       for(Node *pi=l;pi->next!=null;pi=pi->next){
           int min = pi->next->data;
           Node *pre = pi;
           for(Node *pj=pi;pj->next!=null;pj=pj->next){
               if(pj->next->data<min){
                   min = pj->next->data;
                   pre = pj;
               }
           }
           Node *tmp = pre->next;
           pre->next = pre->next->next;
           tmp->next=pi->next;
           pi->next=tmp;
       }
   }
   ```
